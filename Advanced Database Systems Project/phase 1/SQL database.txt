
CREATE TABLE [Member] (
	[MemberID] INT IDENTITY,
	[SSN] INT UNIQUE,
	[Name] varchar(255) ,
	[PhoneNumber] INT,
	[Gender] varchar(5) NOT NULL,
	[Nationality] varchar(255),
	[City] varchar(255),
	[Street] varchar(255),
	[BirthDate] DATETIME,
	PRIMARY KEY ([MemberID])
);

CREATE TABLE [Participate] (
	[MEID] INT NOT NULL,
	[TEID] INT NOT NULL,
	PRIMARY KEY ([MEID],[TEID])
);

CREATE TABLE [Dependent] (
	[DependentID] INT NOT NULL,
	[MEID] INT NOT NULL,
	[Gender] varchar(5) NOT NULL,
	[Name] varchar(255) NOT NULL,
	[Relation] varchar(255),
	[BirthDate] DATETIME,
	PRIMARY KEY ([DependentID],[MEID])
);

CREATE TABLE [Team] (
	[TeamID] INT NOT NULL IDENTITY,
	[SPID] INT NOT NULL,
	[Name] VARCHAR(255) NOT NULL,
	[CoachID] INT NOT NULL,
	PRIMARY KEY ([TeamID])
);

CREATE TABLE [Employee] (
	[EmployeeID] INT NOT NULL IDENTITY,
	[Name] varchar(255) NOT NULL,
	[Gender] varchar(5) NOT NULL,
	[PhoneNumber] INT UNIQUE,
	[Salary] FLOAT NOT NULL,
	[StartDate] DATETIME NOT NULL,
	[EndDate] DATETIME NOT NULL,
	[SSN] INT NOT NULL UNIQUE,
	[City] varchar(255) NOT NULL,
	[Street] varchar(255) NOT NULL,
	[subervisor] INT NOT NULL,
	PRIMARY KEY ([EmployeeID])
);

CREATE TABLE [Trip] (
	[TripID] INT NOT NULL IDENTITY,
	[TripDate] DATETIME,
	[Cost] FLOAT,
	[Duration] TIME,
	[Country] varchar(255),
	[City] varchar(255),
	[Street] varchar(255),
	[SupervisorID] INT NOT NULL,
	PRIMARY KEY ([TripID])
);

CREATE TABLE [Travel] (
	[MEID] INT NOT NULL,
	[TRID] INT NOT NULL,
	[Feedback] INT,
	PRIMARY KEY ([MEID],[TRID])
);

CREATE TABLE [Party] (
	[PartyID] INT NOT NULL IDENTITY,
	[PartyDate] DATETIME,
	[Budget] FLOAT,
	[OrganizerID] INT NOT NULL,
	PRIMARY KEY ([PartyID])
);

CREATE TABLE [Training] (
	[TEID] INT NOT NULL,
	[ZOID] INT NOT NULL,
	[Date] DATETIME,
	[StartTime] TIME,
	[EndTime] TIME,
	PRIMARY KEY ([TEID],[ZOID])
);


CREATE TABLE [Zone] (
	[ZoneID] INT NOT NULL IDENTITY,
	PRIMARY KEY ([ZoneID])
);

CREATE TABLE [ZoneSport] (
	[ZOID] INT NOT NULL,
	[SPID] INT NOT NULL,
	PRIMARY KEY ([ZOID],[SPID])
);

CREATE TABLE [Sport] (
	[SportID] INT NOT NULL IDENTITY,
	[Name] VARCHAR(255) NOT NULL UNIQUE,
	PRIMARY KEY ([SportID])
);

CREATE TABLE [Match] (
	[MatchID] INT NOT NULL IDENTITY,
	[TOID] INT NOT NULL,
	[SPID] INT NOT NULL,
	[FTeamID] INT NOT NULL,
	[STeamID] INT NOT NULL,
	[Date] DATETIME NOT NULL,
	[Time] TIME NOT NULL,
	[FPoints] INT NOT NULL,
	[SPoints] INT NOT NULL,
	[ZOID] INT NOT NULL,
	PRIMARY KEY ([MatchID],[TOID],[SPID])
);

CREATE TABLE [Compete] (
	[TOID] INT NOT NULL,
	[TEID] INT NOT NULL,
	[Standing] INT NOT NULL,
	[Points] INT NOT NULL,
	PRIMARY KEY ([TOID],[TEID])
);

CREATE TABLE [Tournament] (
	[TournamentID] INT NOT NULL IDENTITY,
	[SPID] INT NOT NULL,
	[Start Date] DATETIME,
	PRIMARY KEY ([TournamentID])
);
